ZORUNLU ALAN VALIDATION (Faz 1) ═══════════════════════════════════════════════════════════════════════════ */ // Zorunlu alan ID listesi ve doğrulama tipi window._f1_zorunluAlanlar = [ { id: 'calc-fatura', tip: 'sayi', etiket: 'Fatura Tutarı', minDeger: 1 }, { id: 'calc-yakit', tip: 'select', etiket: 'Yakıt Tipi' }, { id: 'calc-arac-cinsi', tip: 'select', etiket: 'Araç Cinsi' }, { id: 'calc-motor', tip: 'select', etiket: 'Motor Hacmi' }, { id: 'calc-tse-muh', tip: 'sayi', etiket: 'TSE Müh', minDeger: 0 }, { id: 'calc-tse-harc', tip: 'sayi', etiket: 'TSE Harcı', minDeger: 0 }, { id: 'calc-yurt-ici', tip: 'sayi', etiket: 'Yurt İçi', minDeger: 0 }, { id: 'calc-yurt-disi', tip: 'sayi', etiket: 'Yurt Dışı', minDeger: 0 } ]; // Tek bir alanı doğrula window._f1_alanDogrula = function(id) { const el = document.getElementById(id); if (!el) return null; const tanim = window._f1_zorunluAlanlar.find(a => a.id === id); if (!tanim) return null; const deger = el.value; const bos = (deger === '' || deger === null || deger === undefined); let durum = 'bos'; if (!bos) { if (tanim.tip === 'sayi') { const sayi = parseFloat(String(deger).replace(/\./g, '').replace(',', '.')); if (isNaN(sayi)) { durum = 'hatali'; } else if (tanim.minDeger !== undefined && sayi < tanim.minDeger) { durum = sayi === 0 && tanim.minDeger === 0 ? 'dolu' : 'hatali'; } else { durum = 'dolu'; } } else if (tanim.tip === 'select') { durum = (deger && deger !== '') ? 'dolu' : 'bos'; } else { durum = 'dolu'; } } // Önce tüm sınıfları temizle el.classList.remove('zorunlu-bos', 'zorunlu-dolu', 'zorunlu-hatali'); // Yeni durumu ekle if (durum === 'bos') el.classList.add('zorunlu-bos'); else if (durum === 'dolu') el.classList.add('zorunlu-dolu'); else if (durum === 'hatali') el.classList.add('zorunlu-hatali'); return durum; }; // Tüm zorunlu alanları doğrula window._f1_tumAlanlariDogrula = function() { window._f1_zorunluAlanlar.forEach(a => window._f1_alanDogrula(a.id)); }; // Zorunlu alanlara dinleyici bağla window._f1_validationKur = function() { window._f1_zorunluAlanlar.forEach(a => { const el = document.getElementById(a.id); if (!el) return; if (el.dataset.f1Bound === 'true') return; el.dataset.f1Bound = 'true'; const tetikle = () => window._f1_alanDogrula(a.id); el.addEventListener('input', tetikle); el.addEventListener('change', tetikle); el.addEventListener('blur', tetikle); }); // İlk yüklemede tüm alanları kontrol et window._f1_tumAlanlariDogrula(); }; // renderView wrapper'ına validation kurmayı ekle const _orig_renderViewForVal = window.renderView; window.renderView = function(viewName) { let r; if (typeof _orig_renderViewForVal === 'function') { r = _orig_renderViewForVal.apply(this, arguments); } if (viewName === 'calculator') { setTimeout(() => window._f1_validationKur(), 250); } return r; }; // İlk açılışta da kur setTimeout(() => window._f1_validationKur(), 1500); console.log("[Faz1] Zorunlu alan validation kuruldu");
